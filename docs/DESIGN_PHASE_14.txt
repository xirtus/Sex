#### 2. The `sexc` Trampoline (User-Space POSIX Bridge)
**`servers/sexc/src/trampoline.rs`**
This is the background thread spawned inside every protection domain linked against `sexc`. It spin-waits (or yields) on the control ring and directly executes the user's registered C handler.

```rust
use core::sync::atomic::{AtomicPtr, Ordering};
use xipc::spsc::SpscRingBuffer;
use xipc::messages::MessageType;

// Array of 32 atomic pointers to user-registered signal handlers.
// Initialized to null.
static SIGNAL_HANDLERS: [AtomicPtr<()>; 32] = 
    [const { AtomicPtr::new(core::ptr::null_mut()) }; 32];

/// Called by the user-space C library implementation of `signal(2)` or `sigaction(2)`.
#[no_mangle]
pub extern "C" fn sys_sigaction(signo: u32, handler: *mut ()) -> i32 {
    if signo >= 32 {
        return -1; // EINVAL
    }
    SIGNAL_HANDLERS[signo as usize].store(handler, Ordering::Release);
    0
}

/// The entry point for the lock-free background signal thread.
#[no_mangle]
pub extern "C" fn trampoline_entry(control_ring_ptr: *mut SpscRingBuffer) -> ! {
    let ring = unsafe { &*control_ring_ptr };

    loop {
        if let Some(msg) = ring.dequeue() {
            match msg.msg_type {
                MessageType::Signal { signo, sender_capability_id: _ } => {
                    dispatch_signal(signo);
                },
                // Handle other control-plane messages (e.g., capability revocation)
                _ => {} 
            }
        } else {
            // FLSCHED yield: Give up the CPU core if the ring is empty.
            // In a mature state, we would use an MWAIT or similar hardware pause here.
            unsafe { core::arch::asm!("pause") }; // fallback for now
            // sys_yield(); // Call into kernel scheduler
        }
    }
}

fn dispatch_signal(signo: u32) {
    if signo >= 32 { return; }
    
    let handler = SIGNAL_HANDLERS[signo as usize].load(Ordering::Acquire);
    
    if !handler.is_null() {
        // Transmute the raw pointer to a C function pointer and execute it.
        // We are already running in our own trampoline thread context, 
        // so we don't smash the main thread's stack.
        let func: extern "C" fn(u32) = unsafe { core::mem::transmute(handler) };
        func(signo);
    } else {
        // Default actions
        match signo {
            2 | 9 | 15 => { // SIGINT, SIGKILL, SIGTERM
                // sys_exit(128 + signo);
            },
            _ => {} // Ignore others by default
        }
    }
}
```

#### 3. Kernel Router: Translating `kill()` to XIPC
**`kernel/src/xipc/router.rs`**
When the TTY driver (or another PD) wants to send a signal, it doesn't issue a traditional kernel interrupt to the target. It asks the kernel to route a capability-checked `MessageType::Signal`.

```rust
use xipc::messages::{PdxMessage, MessageType};
use crate::scheduler::pd_registry;
use crate::capabilities::CapEngine;

/// Kernel implementation of the signal routing.
/// Returns 0 on success, or a negative error code (e.g., -EPERM).
pub fn route_signal(target_pd: u64, signo: u32, sender_cap_id: u64) -> i32 {
    // 1. Hardware/Software Capability Check
    // Does the sender actually hold a capability allowing it to signal this PD?
    if !CapEngine::verify_signal_rights(sender_cap_id, target_pd) {
        return -1; // EPERM
    }

    // 2. Locate the target Protection Domain's control ring
    let target_ring = match pd_registry::get_control_ring(target_pd) {
        Some(ring) => ring,
        None => return -3, // ESRCH (No such process)
    };

    // 3. Mint the message
    let msg = PdxMessage {
        target_pd,
        msg_type: MessageType::Signal {
            signo,
            sender_capability_id: sender_cap_id,
        },
    };

    // 4. Zero-copy lockless enqueue
    if target_ring.enqueue(msg).is_err() {
        return -11; // EAGAIN (Ring buffer full)
    }

    // 5. Hint to FLSCHED to wake the target's trampoline thread if it's sleeping
    crate::scheduler::wake_thread(target_pd, crate::scheduler::ThreadType::Trampoline);

    0
}
```
